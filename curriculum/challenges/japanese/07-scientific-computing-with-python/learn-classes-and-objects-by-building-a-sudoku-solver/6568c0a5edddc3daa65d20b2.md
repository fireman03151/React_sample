        for index, line in enumerate(self.board):
            row_list = []

                else:
                    board_string += middle_lines
            else:
                board_string += lower_lines

        return board_string

        for row, contents in enumerate(self.board):

                col = contents.index(0)

        return None
    
    def valid_in_row(self, row, num):
